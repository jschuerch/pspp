Object subclass: #Montecarlo
	instanceVariableNames: 'n'
	classVariableNames: ''
	package: 'Montecarlo'




count: anObject
	n := anObject.






pi
	"Approximate pi"

	| redPoints iter rand x y approxpi |
	redPoints := 0.
	iter := n.
	rand := Random new.
	
	1 to: iter do: [ :i |
			x := (rand next).
			y := (rand next).
			(x squared + y squared) < 1
				ifTrue: [ redPoints := redPoints + 1 ] ].
	
	approxpi := (redPoints / iter) asFloat * 4.
	"Transcript 
		show: iter;
		cr;
		show: approxpi; 
		cr."
	^approxpi
